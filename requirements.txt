# Requirements to run this repo
#
# This is a static HTML/CSS/JS site with no build step and no package
# dependencies (no npm/pip packages are installed or required). This file
# lists what needs to be present on your machine to clone and run it.

git                     # to clone the repo
a static file server    # e.g. Python 3 (`python3 -m http.server`) or Node.js (`npx serve .`)
                        # required because the dashboards load data/trades.csv via fetch(),
                        # which browsers block on the file:// scheme
a modern web browser    # Chrome, Firefox, Safari, or Edge, with JavaScript enabled
